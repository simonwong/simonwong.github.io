  private v: number

  private w: number

  weight() {
    return this._weight
  }

  either() {
    return this.v
  }

  other(vertex: number) {
    if (vertex === this.v) {
      return this.w
    }
    if (vertex === this.w) {
      return this.v
    }
    throw Error()
  }

  constructor(v: number, w: number, weight: number) {
    this.v = v
    this.w = w
    this._weight = weight
  }

  compareTo(that: Edge) {
    if (this.weight() < that.weight()) {
      return -1
    }
    if (this.weight() > that.weight()) {
      return 1
    }
    return 0
  }
