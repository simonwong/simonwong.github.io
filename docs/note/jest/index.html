测试 console.log​

let originalLog = null

beforeEach(() => {
  originalLog = global.console.log
  global.console.log = jest.fn()
})

afterEach(() => {
  global.console.log = originalLog
})

// demo
test('console.log', () => {
  console.log('123')
  expect(global.console.log).toHaveBeenCalledWith('123')
})
