function currying (fn) {
  return function curried (...args) {
    if (args.length >= fn.length) {
      return fn.apply(null, args)
    } else {
      return (...restArgs) => {
        return curried.apply(null, args.concat(restArgs))
      }
    }
  }
}
const sum = (a, b, c) => {
  return a + b + c
}

const curriedSum = currying(sum)

curriedSum(1, 2)(3) // => 6
curriedSum(1)(2)(3) // => 6
