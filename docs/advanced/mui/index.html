useEventCallback​

可以用于事件绑定函数的存储

/**
 * https://github.com/facebook/react/issues/14099#issuecomment-440013892
 */
function useEventCallback<Args extends unknown[], Return>(
  fn: (...args: Args) => Return,
): (...args: Args) => Return {
  const ref = React.useRef(fn);
  useEffectLayout(() => {
    ref.current = fn;
  });

  return React.useCallback((...args: Args) => (0, ref.current!)(...args), []);
}
