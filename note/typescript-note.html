# TypeScript 笔记

# 技巧

# 以对象的 key 作为联合类型

const typeMap = {
    1: 'a',
    2: 'c',
    3: 'g',
}

type Typekey = keyof typeof typeMap

# 以对象的 value 作为联合类型

type ValueOf<T> = T[keyof T]

const typeMap = {
    1: 'a',
    2: 'c',
    3: 'g',
}

type TypeValue = ValueOf<typeof typeMap>

# 以对象的 value 值最为类型

// ts 会默认推导类型 `white` 的类型为 `string`
const color = {
  white: '#fff',
  black: '#000'
}

改成

// 此时 white 的类型为 ‘#fff’
const color = {
  white: '#fff',
  black: '#000'
} as const

# ESNext 编译成 CommonJS

// tsconfig.json
{
  "compilerOptions": {
    "module": "CommonJS",
    "target": "ES2015"
  },
}

# ts 检测三方库 types 报错

// tsconfig.json
{
  "compilerOptions": {
    "skipLibCheck": true
  },
}
